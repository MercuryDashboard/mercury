Template - Import template from Link tag Allow developer to Import HTML Template. Clone for using. 

					/*
						Two ways to import HTML Web Component - Template
					 */
					
					// First to declare which page to Import
					<link id="template-import" rel="import" href="path/template.html">

					// Import template (Shadow Dom) via JavaScript
					var import = document.querySelector("#template-import").import,
						target = import.querySelector('template').cloneNode(true);
						
					// Append Template-Target to Dom-body
					document.querySelector('body').appendChild(target.content.cloneNode(true));
