Util -> Template Allow developer to Import HTML Template. Clone for using. 

					// First to declare which page to Import
					<link id="template-import" rel="import" href="path/template.html">

					// Import template (Shadow Dom) via JavaScript
					var import = document.querySelector("#template-import").import,
						target = import.querySelector('div');
						
					// Append Template-Target to Dom-body
					document.querySelector('body').appendChild(target.cloneNode(true));
