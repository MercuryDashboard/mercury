Util -> type(o); Allow developer to Identify the type of veriable. Return pure typeof. 

				/*
					Identify specific typeof of element.
					Example for using:
						type(parameter) ----> Will return specific type of this param.
					Return String value if detected:
						'Object', 'Array', 'Boolean', 'Number', 'String' 
				*/
				var type = function(o){
				    return Object.prototype.toString.call(o).replace(/\[object +(.*?)\]+/g,function(s0,s1){
				           return s1;
				    });
				};
